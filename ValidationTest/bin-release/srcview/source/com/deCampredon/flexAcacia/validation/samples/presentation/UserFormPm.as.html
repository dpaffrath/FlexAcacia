package com.deCampredon.flexAcacia.validation.samples.presentation
{
    import com.deCampredon.flexAcacia.validation.ValidationSessionFactory;
    import com.deCampredon.flexAcacia.validation.samples.vo.User;

    [Bindable]
Assert("passwordConfirm",expression="confirm == password",errorMessage="password and confirmation must be equals")]
    public class UserFormPm
    {
        
        
        
        public function UserFormPm()
        {
            user = new User();
            validator = new ValidationSessionFactory(UserFormPm);
        }
        
        public var validator:ValidationSessionFactory;
        
        [SubValidator]public var user:User;
        
        [NotNull][Size("8")]
        [Pattern(".*[a-z].*",flags="i",noMatchError="password must contain letter")]
        [Pattern(".*[1-9].*",flags="i",noMatchError="password must contain digit")]
        public var password:String;
        
        public var confirm:String;
    }
}
